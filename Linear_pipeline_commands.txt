#FastQC
fastqc -o QC/fastqc {input}

#MultiQC
multiqc -o QC/multiqc QC/fastqc

#bwa_map
bwa mem -M -p -t {threads} {input} | samtools view -b -h -@ {threads}  > {output}

#AddOrReplaceReadGroups
picard AddOrReplaceReadGroups I={input} O={output} RGID=1 RGLB=lib2 RGPL=ILLUMINA RGPU=unit1 RGSM={wildcards.sample}

#SortSam
picard -Xmx13g SortSam I={input} O={output} VALIDATION_STRINGENCY=LENIENT SORT_ORDER=coordinate MAX_RECORDS_IN_RAM=3000000 CREATE_INDEX=True

#MarkDuplicates
picard -Xmx13g MarkDuplicates I={input} O={output.bam} METRICS_FILE={output.met}

#BaseRecalibrator
gatk --java-options -Xmx13g BaseRecalibrator -I {input.bam} -R /root/Project/data/Homo_sapiens_assembly38.fasta --known-sites /root/Project/data/00-All.vcf.gz -O {output}

#ApplyBQSR
gatk --java-options -Xmx13g ApplyBQSR -I {input.bam} -R /root/Project/data/Homo_sapiens_assembly38.fasta --bqsr-recal-file {input.recal} -O {output}

#HaplotypeCaller
gatk --java-options -Xmx13g HaplotypeCaller -I {input.gatk} -R /root/Project/data/Homo_sapiens_assembly38.fasta -ERC GVCF -O {output}

#gatk_CombineGVCFs
ls gvcf/*.gz >gvcfs.list && gatk --java-options -Xmx13g CombineGVCFs -R /root/Project/data/Homo_sapiens_assembly38.fasta --variant gvcfs.list -O {output.gvcf}

#GenotypeGVCFs
gatk --java-options -Xmx13g GenotypeGVCFs -R /root/Project/data/Homo_sapiens_assembly38.fasta -V /root/docker/gvcf_combined/all.g.vcf.gz -O {output}

#LeftAlignAndTrimVariants
gatk --java-options -Xmx13g LeftAlignAndTrimVariants -R /root/Project/data/Homo_sapiens_assembly38.fasta -V {input.vcf} -O {output} --split-multi-allelics

#SelectVariants_SNPS
gatk --java-options -Xmx13g SelectVariants -V {input} -select-type SNP -O {output}

#SelectVariants_INDELS
gatk --java-options -Xmx13g SelectVariants -V {input} -select-type INDEL -O {output}

#VariantFiltration_SNP
gatk --java-options -Xmx13g VariantFiltration -V {input} \
-filter "QD < 2.0" --filter-name "QD2" \
-filter "QUAL < 30.0" --filter-name "QUAL30" \
-filter "SOR > 3.0" --filter-name "SOR3" \
-filter "FS > 60.0" --filter-name "FS60" \
-filter "MQ < 40.0" --filter-name "MQ40" \
-filter "MQRankSum < -12.5" --filter-name "MQRankSum-12.5" \
-filter "ReadPosRankSum < -8.0" --filter-name "ReadPosRankSum-8 \
-O {output}
                    
#VariantFiltration_INDELS 
gatk --java-options -Xmx13g VariantFiltration -V {input} \
-filter "QD < 2.0" --filter-name "QD2" \
-filter "QUAL < 30.0" --filter-name "QUAL30" \
-filter "FS > 200.0" --filter-name "FS200" \
-filter "ReadPosRankSum < -20.0" --filter-name "ReadPosRankSum-20 \
-O {output}
